================================================================================
  1. OVERVIEW
================================================================================

EZQueue is a cross-platform mobile queueing system built with Flutter and Dart.
This guide defines the complete design system — from color tokens and typography
to component behavior and responsive layout rules — ensuring a consistent user
experience across all mobile phones and tablets.

PROJECT STACK
--------------------------------------------------------------------------------
Layer                   Technology
--------------------------------------------------------------------------------
Framework               Flutter 3.x (Dart 3.x)
State Management        Riverpod / Bloc
Navigation              GoRouter
Minimum SDK             Android 5.0 (API 21) / iOS 13
Tablet Support          Adaptive layouts via LayoutBuilder
Font Package            google_fonts (Inter + JetBrains Mono)
--------------------------------------------------------------------------------

  2. COLOR SYSTEM
================================================================================

The recommended default is the Pure & Bold theme (Option 3), which serves as

================================================================================
  4. THEMEDATA SETUP IN FLUTTER
================================================================================

All five themes are implemented as ThemeData objects. The app uses a ThemeMode
enum to switch between light and dark, and a custom AppTheme class to select
between the five personality themes.


4.1 THEME STRUCTURE
--------------------------------------------------------------------------------

// lib/theme/app_theme.dart
enum AppThemeVariant { pureBold, techy, friendly, corporate, playful }

class AppTheme {
  static ThemeData light(AppThemeVariant v) {
    final colors = _lightColors(v);
    return ThemeData(
      useMaterial3: true,
      colorScheme: ColorScheme.light(
        background:  colors.background,
        primary:     colors.primary,
        secondary:   colors.accent,
        surface:     colors.surface,
      ),
      textTheme: EZTypography.textTheme,
      scaffoldBackgroundColor: colors.background,
      appBarTheme: AppBarTheme(
        backgroundColor: colors.background,
        foregroundColor: colors.primary,
        elevation: 0,
      ),
    );
  }

  static ThemeData dark(AppThemeVariant v) {
    final colors = _darkColors(v);
    return ThemeData(
      useMaterial3: true,
      brightness: Brightness.dark,
      colorScheme: ColorScheme.dark(
        background:  colors.background,
        primary:     colors.primary,
        secondary:   colors.accent,
        surface:     colors.surface,
      ),
      textTheme: EZTypography.textTheme,
    );
  }
}


================================================================================
  5. SPACING & LAYOUT SYSTEM
================================================================================

EZQueue uses an 8pt base grid. All spacing, padding, and margin values must
be multiples of 4.

--------------------------------------------------------------------------------

--------------------------------------------------------------------------------

space8      8px     Card internal padding (small)
space12     12px    Chip padding, small gaps
space16     16px    Standard padding, list item spacing
space24     24px    Card padding, section separation
space32     32px    Large section gaps
space48     48px    Hero sections, screen top padding
space64     64px    Extra large spacing
--------------------------------------------------------------------------------

// lib/theme/spacing.dart
class EZSpacing {
  static const double xs    = 4;
  static const double sm    = 8;
  static const double md    = 16;
  static const double lg    = 24;
  static const double xl    = 32;
  static const double xxl   = 48;
  static const double xxxl  = 64;

  // Border radii
  static const double radiusSm   = 6;
  static const double radiusMd   = 12;
  static const double radiusLg   = 20;
  static const double radiusFull = 999;
}


5.2 RESPONSIVE BREAKPOINTS
--------------------------------------------------------------------------------
Breakpoint  Width       Layout
--------------------------------------------------------------------------------
compact     < 600px     Phone portrait — single column
medium      600-840px   Phone landscape / small tablet — two columns
expanded    > 840px     Tablet — sidebar + content layout
--------------------------------------------------------------------------------

// lib/utils/responsive.dart
class Responsive {
  static bool isCompact(BuildContext ctx) =>
    MediaQuery.of(ctx).size.width < 600;

  static bool isMedium(BuildContext ctx) {
    final w = MediaQuery.of(ctx).size.width;
    return w >= 600 && w < 840;
  }

  static bool isExpanded(BuildContext ctx) =>
    MediaQuery.of(ctx).size.width >= 840;

  static int gridColumns(BuildContext ctx) {
    if (isExpanded(ctx)) return 3;
    if (isMedium(ctx))   return 2;
    return 1;
  }
}

// lib/widgets/queue_ticket_card.dart
class QueueTicketCard extends StatelessWidget {
  final String queueNumber;
  final String status;   // 'queued' | 'serving' | 'done'
  final int estimatedMinutes;
  final String counterName;

  const QueueTicketCard({
    required this.queueNumber,
    required this.status,
    required this.estimatedMinutes,
    required this.counterName,
    super.key,
  });

  Color _statusColor(BuildContext ctx) {
    return switch(status) {
      'serving' => const Color(0xFF10B981),
      'queued'  => const Color(0xFF8B5CF6),
      _         => const Color(0xFF6B7280),
    };
  }

  @override
  Widget build(BuildContext context) {
    return Card(
      elevation: 2,
      shape: RoundedRectangleBorder(
        borderRadius: BorderRadius.circular(EZSpacing.radiusMd),
      ),
      child: Padding(
        padding: const EdgeInsets.all(EZSpacing.lg),
        child: Column(children: [
          Text(queueNumber, style: EZTypography.queueNumber),
          StatusBadge(status: status, color: _statusColor(context)),
          Text('~$estimatedMinutes min wait'),
          Text(counterName),
        ]),
      ),
    );
  }
}

7.2 TABLET NAVIGATION — Side Rail

On tablets (expanded breakpoint), replace BottomNavigationBar with a
NavigationRail or NavigationDrawer on the left. Content takes the remaining
space.

// lib/shell/app_shell.dart
Widget build(BuildContext context) {
  final isTablet = Responsive.isExpanded(context);
  return Scaffold(
    body: Row(
      children: [
        if (isTablet)
          NavigationRail(
            destinations: navDestinations,
            selectedIndex: currentIndex,
            onDestinationSelected: (i) => onNavTap(i, context),
            labelType: NavigationRailLabelType.all,
          ),
        Expanded(child: child),  // GoRouter outlet
      ],
    ),
    bottomNavigationBar: isTablet ? null : BottomNavigationBar(
      items: navDestinations,
      currentIndex: currentIndex,
      onTap: (i) => onNavTap(i, context),
    ),
  );
}

================================================================================
  11. THEME SWITCHING & PERSISTENCE
================================================================================

Users can switch the personality theme and brightness mode from Settings.
Preferences are persisted using shared_preferences.

// lib/providers/theme_provider.dart  (Riverpod)
final themeVariantProvider =
  StateNotifierProvider<ThemeNotifier, AppThemeVariant>((ref) {
    return ThemeNotifier();
  });

class ThemeNotifier extends StateNotifier<AppThemeVariant> {
  ThemeNotifier() : super(AppThemeVariant.pureBold) {
    _loadSavedTheme();
  }

  Future<void> setTheme(AppThemeVariant v) async {
    state = v;
    final prefs = await SharedPreferences.getInstance();
    await prefs.setString('app_theme', v.name);
  }

  Future<void> _loadSavedTheme() async {
    final prefs = await SharedPreferences.getInstance();
    final saved = prefs.getString('app_theme');
    if (saved != null) {
      state = AppThemeVariant.values.firstWhere(
        (v) => v.name == saved,
        orElse: () => AppThemeVariant.pureBold,
      );
    }
  }
}
